export const exampleCode = `
import { connect, unlockMedal, postScore, login } from 'newgrounds.js';

// Connecting to Newgrounds.io
connect('your-game-id', 'your-encryption-key');

// Logging in
login();

// Unlocking a medal
unlockMedal('medal-id');

// Posting a score
postScore('scoreboard-id', 100);

`
